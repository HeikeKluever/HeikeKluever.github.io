\DeclareBibliographyDriver{article}{%
	\usebibmacro{bibindex}%
	\usebibmacro{begentry}%
	\usebibmacro{author}%
	\setunit{\addcolon\space}%
	\newblock
	\printfield{title}%
	\newunit\newblock
	\printfield{journaltitle}%
	\setunit{\addspace}%
	\printfield{volume}%
	\setunit{\addspace}%
	\printfield{number}%
	\setunit{\addcomma\space}%
	\printfield{pages}%
	\newunit\newblock
	\setunit{\addperiod\addspace}%
	\printfield{doi}%
	\newunit\newblock
	\usebibmacro{finentry}%
}

	\ifnameundef{editor}
	{}
	{\printnames{editor}%
		\setunit{\addspace}%
		\ifnumgreater{\value{editor}}{1}
		{\printtext{(eds.):}}%
		{\printtext{(ed.):}}}}%
